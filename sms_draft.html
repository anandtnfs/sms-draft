    document.getElementById("createSmsDraft").addEventListener("click", function() {

      const phoneNumber = "1234567890"; // Replace with the desired phone number

      const messageBody = "Hello, this is a draft message."; // Replace with the desired message body

      const encodedMessageBody = encodeURIComponent(messageBody);

      const smsLink = `sms:${phoneNumber}&body=${encodedMessageBody}`;


      window.location.href = smsLink;

    });
